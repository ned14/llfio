132  // Holds state for an i/o in progress. Will be subclassed with platform specific state and how to implement completion.
133  // Note this is allocated using malloc not new to avoid memory zeroing, and therefore it has a custom deleter.
134  struct _erased_io_state_type
135  {
136  async_file_handle *parent;
137  operation_t operation;
138  size_t items;
139  shared_size_type items_to_go;
140  constexpr _erased_io_state_type(async_file_handle *_parent, operation_t _operation, size_t _items)
141  : parent(_parent)
142  , operation(_operation)
143  , items(_items)
144  , items_to_go(0)
145  {
146  }
147  /*
148  For Windows:
149  - errcode: GetLastError() code
150  - bytes_transferred: obvious
151  - internal_state: LPOVERLAPPED for this op
152 
153  For POSIX AIO:
154  - errcode: errno code
155  - bytes_transferred: return from aio_return(), usually bytes transferred
156  - internal_state: address of pointer to struct aiocb in io_service's _aiocbsv
157  */
158  virtual void operator()(long errcode, long bytes_transferred, void *internal_state) noexcept = 0;
159  BOOST_AFIO_HEADERS_ONLY_VIRTUAL_SPEC ~_erased_io_state_type()
160  {
161  // i/o still pending is very bad, this should never happen
162  assert(!items_to_go);
163  if(items_to_go)
164  {
165  BOOST_AFIO_LOG_FATAL(parent->native_handle().h, "FATAL: io_state destructed while i/o still in flight, the derived class should never allow this.");
166  abort();
167  }
168  }
169  };
