188  static result<memory_map> fs_mutex_map(file_handle::path_type lockfile, shared_fs_mutex *fallbacklock = nullptr) noexcept
189  {
190  BOOST_AFIO_LOG_FUNCTION_CALL(0);
191  try
192  {
193  BOOST_OUTCOME_FILTER_ERROR(ret, file_handle::file(std::move(lockfile), file_handle::mode::write, file_handle::creation::if_needed, file_handle::caching::temporary, file_handle::flag::win_delete_on_last_close));
194  file_handle temph;
195  // Am I the first person to this file? Lock the inuse exclusively
196  auto lockinuse = ret.try_lock(_lockinuseoffset, 1, true);
198  file_handle::extent_guard mapinuse;
199  if(lockinuse.has_error())
200  {
201  if(lockinuse.get_error().value() != ETIMEDOUT)
202  return lockinuse.get_error();
203  // Somebody else is also using this file, so try to read the hash index file I ought to use
204  lockinuse = ret.lock(_lockinuseoffset, 1, false); // last byte shared access
205  char buffer[65536];
206  memset(buffer, 0, sizeof(buffer));
207  {
208  BOOST_OUTCOME_FILTER_ERROR(_, ret.read(0, buffer, 65535));
209  (void) _;
210  }
211  fixme_path::value_type *temphpath = (fixme_path::value_type *) buffer;
212  result<file_handle> _temph;
213  // If path is zeroed, fall back onto backup lock
214  if(!buffer[1])
215  goto use_fall_back_lock;
216  else
217  _temph = file_handle::file(temphpath, file_handle::mode::write, file_handle::creation::open_existing, file_handle::caching::temporary, file_handle::flag::win_delete_on_last_close);
218  // If temp file doesn't exist, I am on a different machine
219  if(!_temph)
220  {
221  // Zop the path so any new entrants into this lock will go to the fallback lock
222  memset(buffer, 0, 4096);
223  (void) ret.write(0, buffer, 4096);
224  use_fall_back_lock:
225  // I am guaranteed that all mmap users have locked the second last byte
226  // and will unlock it once everyone has stopped using the mmap, so make
227  // absolutely sure the mmap is not in use by anyone by taking an exclusive
228  // lock on the second final byte
229  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, ret.lock(_mapinuseoffset, 1, true));
230  // Release the exclusive lock and tell caller to just use the fallback lock directly
231  return make_errored_result<memory_map>(EBUSY);
232  }
233  else
234  {
235  // Mark the map as being in use by me too
236  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, ret.lock(_mapinuseoffset, 1, false));
237  mapinuse = std::move(mapinuse2);
238  temph = std::move(_temph.get());
239  }
240  }
241  else
242  {
243  // I am the first person to be using this (stale?) file, so create a new hash index file and write its path
244  ret.truncate(0);
245  BOOST_OUTCOME_FILTER_ERROR(_temph, file_handle::temp_file());
246  temph = std::move(_temph);
247  auto temppath(temph.path());
248  temph.truncate(HashIndexSize);
249  // Write the path of my new hash index file and convert my lock to a shared one
250  {
251  BOOST_OUTCOME_FILTER_ERROR(_, ret.write(0, (const char *) temppath.c_str(), temppath.native().size() * sizeof(*temppath.c_str())));
252  (void) _;
253  }
254  // Convert exclusive whole file lock into lock in use
255  BOOST_OUTCOME_FILTER_ERROR(lockinuse2, ret.lock(_lockinuseoffset, 1, false));
256  BOOST_OUTCOME_FILTER_ERROR(mapinuse2, ret.lock(_mapinuseoffset, 1, false));
257  mapinuse = std::move(mapinuse2);
258  lockinuse = std::move(lockinuse2);
259  }
260  // Map the files into memory, being very careful that the lock file is only ever mapped read only
261  // as some OSs can get confused if you use non-mmaped writes on a region mapped for writing.
262  BOOST_OUTCOME_FILTER_ERROR(hsection, section_handle::section(ret, 0, section_handle::flag::read));
263  BOOST_OUTCOME_FILTER_ERROR(temphsection, section_handle::section(temph, HashIndexSize));
264  BOOST_OUTCOME_FILTER_ERROR(hmap, map_handle::map(hsection, 0, 0, section_handle::flag::read));
265  BOOST_OUTCOME_FILTER_ERROR(temphmap, map_handle::map(temphsection, HashIndexSize));
266  return memory_map(std::move(ret), std::move(temph), std::move(lockinuse.get()), std::move(mapinuse), std::move(hmap), std::move(temphmap), fallbacklock);
267  }
268  BOOST_OUTCOME_CATCH_EXCEPTION_TO_RESULT(memory_map)

280  // Create a cache of entities to their indices, eliding collisions where necessary
281  static span<_entity_idx> _hash_entities(_entity_idx *entity_to_idx, entities_type &entities)
282  {
283  _entity_idx *ep = entity_to_idx;
284  for(size_t n = 0; n < entities.size(); n++)
285  {
286  ep->value = hasher_type()(entities[n].value) % _container_entries;
287  ep->exclusive = entities[n].exclusive;
288  bool skip = false;
289  for(size_t m = 0; m < n; m++)
290  {
291  if(entity_to_idx[m].value == ep->value)
292  {
293  if(ep->exclusive && !entity_to_idx[m].exclusive)
294  entity_to_idx[m].exclusive = true;
295  skip = true;
296  }
297  }
298  if(!skip)
299  ++ep;
300  }
301  return span<_entity_idx>(entity_to_idx, ep - entity_to_idx);
302  }
303  virtual result<void> _lock(entities_guard &out, deadline d, bool spin_not_sleep) noexcept override final
304  {
305  BOOST_AFIO_LOG_FUNCTION_CALL(this);
306  if(is_degraded())
307  {
308  _hmapinuse.unlock();
309  if(_fallbacklock)
310  return _fallbacklock->_lock(out, d, spin_not_sleep);
311  return make_errored_result<void>(EBUSY);
312  }
313  stl11::chrono::steady_clock::time_point began_steady;
314  stl11::chrono::system_clock::time_point end_utc;
315  if(d)
316  {
317  if((d).steady)
318  began_steady = stl11::chrono::steady_clock::now();
319  else
320  end_utc = (d).to_time_point();
321  }
322  span<_entity_idx> entity_to_idx(_hash_entities((_entity_idx *) alloca(sizeof(_entity_idx) * out.entities.size()), out.entities));
323  _hash_index_type &index = _index();
324  // Fire this if an error occurs
325  auto disableunlock = detail::Undoer([&] { out.release(); });
326  size_t n;
327  for(;;)
328  {
329  size_t was_contended = (size_t) -1;
330  {
331  auto undo = detail::Undoer([&] {
332  // 0 to (n-1) need to be closed
333  if(n > 0)
334  {
335  --n;
336  // Now 0 to n needs to be closed
337  for(; n > 0; n--)
338  entity_to_idx[n].exclusive ? index[entity_to_idx[n].value].unlock() : index[entity_to_idx[n].value].unlock_shared();
339  entity_to_idx[0].exclusive ? index[entity_to_idx[0].value].unlock() : index[entity_to_idx[0].value].unlock_shared();
340  }
341  });
342  for(n = 0; n < entity_to_idx.size(); n++)
343  {
344  if(!(entity_to_idx[n].exclusive ? index[entity_to_idx[n].value].try_lock() : index[entity_to_idx[n].value].try_lock_shared()))
345  {
346  was_contended = n;
347  goto failed;
348  }
349  }
350  // Everything is locked, exit
351  undo.dismiss();
352  disableunlock.dismiss();
353  return make_result<void>();
354  }
355  failed:
356  if(d)
357  {
358  if((d).steady)
359  {
360  if(stl11::chrono::steady_clock::now() >= (began_steady + stl11::chrono::nanoseconds((d).nsecs)))
361  return make_errored_result<void>(ETIMEDOUT);
362  }
363  else
364  {
365  if(stl11::chrono::system_clock::now() >= end_utc)
366  return make_errored_result<void>(ETIMEDOUT);
367  }
368  }
369  // Move was_contended to front and randomise rest of out.entities
370  std::swap(entity_to_idx[was_contended], entity_to_idx[0]);
371  auto front = entity_to_idx.begin();
372  ++front;
373  std::random_shuffle(front, entity_to_idx.end());
374  if(!spin_not_sleep)
375  std::this_thread::yield();
376  }
377  // return make_result<void>();
378  }
