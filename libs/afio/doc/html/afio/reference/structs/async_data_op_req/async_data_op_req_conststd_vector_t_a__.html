async_data_op_req< const std::vector< T, A > >

A convenience bundle of precondition, data and where for writing from a const std::vector<T, A>. Data MUST stay around until the operation completes.

Synopsis 
template<class T, class A>
struct async_data_op_req< const std::vector< T, A > >
      : public detail::async_data_op_req_impl< true >
{
  std::vector< boost::asio::const_buffer > buffers;  // A sequence of const Boost.ASIO buffers to write from. 
  async_io_op precondition;                          // A precondition containing an open file handle for this operation. 
  off_t where;                                       // The offset at which to write. 
};
