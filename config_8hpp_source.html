620  template <typename callable> class UndoerImpl
621  {
622  bool _dismissed;
623  callable undoer;
624  UndoerImpl() = delete;
625  UndoerImpl(const UndoerImpl &) = delete;
626  UndoerImpl &operator=(const UndoerImpl &) = delete;
627  explicit UndoerImpl(callable &&c)
628  : _dismissed(false)
629  , undoer(std::move(c))
630  {
631  }
632  void int_trigger()
633  {
634  if(!_dismissed && !is_nullptr(undoer))
635  {
636  undoer();
637  _dismissed = true;
638  }
639  }

641  public:

643  UndoerImpl &operator=(UndoerImpl &&o) noexcept
644  {
645  int_trigger();
646  _dismissed = o._dismissed;
647  undoer = std::move(o.undoer);
648  o._dismissed = true;
649  return *this;
650  }
651  template <typename _callable> friend UndoerImpl<_callable> Undoer(_callable c);
652  ~UndoerImpl() { int_trigger(); }
654  bool dismissed() const { return _dismissed; }
656  void dismiss(bool d = true) { _dismissed = d; }
658  void undismiss(bool d = true) { _dismissed = !d; }
659  }; // UndoerImpl
